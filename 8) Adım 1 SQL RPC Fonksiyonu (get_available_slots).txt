-- Eski versiyon varsa temizle
DROP FUNCTION IF EXISTS get_available_slots;

CREATE OR REPLACE FUNCTION get_available_slots(
  p_date DATE,                 -- Örn: '2026-03-03'
  p_staff_id UUID,             -- Personel ID
  p_duration_min INT           -- Hizmet süresi (Örn: 30 dk)
) 
RETURNS TABLE (
  slot_time TEXT,              -- Saat formatı (HH:MM)
  is_available BOOLEAN         -- Müsaitlik durumu
) 
LANGUAGE plpgsql
AS $$
DECLARE
  v_day_name TEXT;             -- 'monday', 'tuesday' vb.
  v_shift_start TEXT;          -- '09:00'
  v_shift_end TEXT;            -- '18:00'
  v_slot_start TIMESTAMPTZ;    -- Döngüdeki anlık slot başlangıcı
  v_slot_end TIMESTAMPTZ;      -- Döngüdeki anlık slot bitişi
  v_work_start_ts TIMESTAMPTZ; -- O günkü mesai başlangıcı (Tarihli)
  v_work_end_ts TIMESTAMPTZ;   -- O günkü mesai bitişi (Tarihli)
  v_interval INTERVAL;         -- Hizmet süresi (Interval tipinde)
  v_timezone TEXT := 'Europe/Istanbul'; -- Zaman dilimi sabit
BEGIN
  -- 1. Günün ismini bul (lowercased: 'monday', 'tuesday'...)
  SELECT lower(trim(to_char(p_date, 'Day'))) INTO v_day_name;

  -- 2. Personelin o günkü çalışma saatlerini JSON'dan çek
  SELECT 
    availability -> v_day_name ->> 0, -- Başlangıç saati (örn: "09:00")
    availability -> v_day_name ->> 1  -- Bitiş saati (örn: "18:00")
  INTO v_shift_start, v_shift_end
  FROM staff
  WHERE id = p_staff_id;

  -- 3. Eğer personel o gün çalışmıyorsa (null ise) boş dön
  IF v_shift_start IS NULL OR v_shift_end IS NULL THEN
    RETURN; -- Hiçbir satır döndürme
  END IF;

  -- 4. String saatleri (09:00) o günün tarihine (Timestamp) çevir
  -- Örn: '2026-03-03 09:00:00+03'
  v_work_start_ts := (p_date || ' ' || v_shift_start)::TIMESTAMP AT TIME ZONE v_timezone;
  v_work_end_ts := (p_date || ' ' || v_shift_end)::TIMESTAMP AT TIME ZONE v_timezone;
  
  -- Dakikayı interval'e çevir
  v_interval := (p_duration_min || ' minutes')::INTERVAL;

  -- 5. Mesai başlangıcından bitişine kadar döngü kur (Slot Hesaplama)
  v_slot_start := v_work_start_ts;

  WHILE v_slot_start + v_interval <= v_work_end_ts LOOP
    v_slot_end := v_slot_start + v_interval;

    -- ÇAKIŞMA KONTROLÜ:
    -- Bu slot aralığında (v_slot_start ile v_slot_end arası)
    -- 'confirmed' statüsünde başka bir randevu var mı?
    IF NOT EXISTS (
      SELECT 1 FROM appointments
      WHERE staff_id = p_staff_id
      AND status = 'confirmed'
      AND tstzrange(start_at, end_at) && tstzrange(v_slot_start, v_slot_end)
    ) THEN
      -- Çakışma yoksa listeye ekle
      slot_time := to_char(v_slot_start AT TIME ZONE v_timezone, 'HH24:MI');
      is_available := true;
      RETURN NEXT;
    END IF;

    -- Bir sonraki slota geç (Örn: 30 dk ekle)
    -- İpucu: Burayı dilerseniz 'duration' kadar değil, sabit 30 dk veya 15 dk artırabilirsiniz.
    -- Şimdilik hizmet süresi kadar atlatıyoruz (Blok Randevu Mantığı).
    v_slot_start := v_slot_start + v_interval; 
  END LOOP;

END;
$$;
