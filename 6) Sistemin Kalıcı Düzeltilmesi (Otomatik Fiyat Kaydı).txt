-- Hatalı fonksiyonu onarıyoruz
CREATE OR REPLACE FUNCTION get_dashboard_stats()
RETURNS jsonb AS $$
DECLARE
  today_count INT;
  monthly_revenue NUMERIC;
  total_customers INT;
BEGIN
  -- 1. Bugünkü Randevular (DÜZELTİLDİ)
  -- Parantez içine alarak önce saat dilimini çevir, sonra tarihe dönüştür diyoruz.
  SELECT COUNT(*) INTO today_count 
  FROM appointments 
  WHERE (start_at AT TIME ZONE 'Europe/Istanbul')::date = CURRENT_DATE;

  -- 2. Bu Ayki Ciro
  SELECT COALESCE(SUM(price_at_booking), 0) INTO monthly_revenue
  FROM appointments
  WHERE status IN ('confirmed', 'completed')
  AND start_at >= date_trunc('month', CURRENT_DATE);

  -- 3. Toplam Müşteri
  SELECT COUNT(*) INTO total_customers FROM customers;

  RETURN jsonb_build_object(
    'today_appointments', today_count,
    'monthly_revenue', monthly_revenue,
    'total_customers', total_customers
  );
END;
$$ LANGUAGE plpgsql;
